# sm121-kernels Design Document

## 1. Build System

### build.rs

The build script is the first thing to implement. It orchestrates:
1. Finding ptxas on the system
2. Discovering all .ptx files under `ptx/`
3. Preprocessing each through `cpp -P` to resolve `#include` directives
4. Assembling each into a cubin via `ptxas -arch=sm_121a`
5. Generating Rust code that embeds all cubins via `include_bytes!`

```rust
// crates/sm121-kernels/build.rs

use std::env;
use std::fs;
use std::io::Write;
use std::path::{Path, PathBuf};
use std::process::Command;

fn main() {
    let out_dir = PathBuf::from(env::var("OUT_DIR").unwrap());
    let manifest_dir = PathBuf::from(env::var("CARGO_MANIFEST_DIR").unwrap());
    let ptx_root = manifest_dir
        .parent()
        .unwrap()
        .parent()
        .unwrap()
        .join("ptx");

    let ptxas = find_ptxas();
    let cpp = find_cpp();
    let ptx_files = discover_ptx_files(&ptx_root);

    let mut kernels: Vec<(String, PathBuf)> = Vec::new();

    for ptx_file in &ptx_files {
        println!("cargo:rerun-if-changed={}", ptx_file.display());

        let stem = ptx_file.file_stem().unwrap().to_str().unwrap().to_string();

        // Step 1: Preprocess (resolve #include directives)
        let preprocessed_path = out_dir.join(format!("{stem}.preprocessed.ptx"));
        let cpp_status = Command::new(&cpp)
            .args([
                "-P",
                "-I",
                ptx_root.join("common").to_str().unwrap(),
                ptx_file.to_str().unwrap(),
                "-o",
                preprocessed_path.to_str().unwrap(),
            ])
            .status()
            .unwrap_or_else(|e| panic!("failed to run cpp: {e}"));

        if !cpp_status.success() {
            panic!("cpp preprocessing failed for {}", ptx_file.display());
        }

        // Step 2: Assemble to cubin
        let cubin_path = out_dir.join(format!("{stem}.cubin"));
        let ptxas_status = Command::new(&ptxas)
            .args([
                "-arch=sm_121a",
                "-O3",
                "--warn-on-spills",
                "-o",
                cubin_path.to_str().unwrap(),
                preprocessed_path.to_str().unwrap(),
            ])
            .status()
            .unwrap_or_else(|e| panic!("failed to run ptxas: {e}"));

        if !ptxas_status.success() {
            panic!("ptxas failed for {}", ptx_file.display());
        }

        kernels.push((stem, cubin_path));
    }

    // Also watch all .ptxh header files for changes
    if let Ok(entries) = fs::read_dir(ptx_root.join("common")) {
        for entry in entries.flatten() {
            let path = entry.path();
            if path.extension().map_or(false, |e| e == "ptxh") {
                println!("cargo:rerun-if-changed={}", path.display());
            }
        }
    }

    // Generate embedded_kernels.rs
    let gen_path = out_dir.join("embedded_kernels.rs");
    let mut f = fs::File::create(&gen_path).unwrap();

    writeln!(f, "// Auto-generated by build.rs. Do not edit.").unwrap();
    writeln!(f, "use std::collections::HashMap;").unwrap();
    writeln!(f, "use std::sync::LazyLock;").unwrap();
    writeln!(f).unwrap();
    writeln!(
        f,
        "pub static KERNEL_REGISTRY: LazyLock<HashMap<&'static str, &'static [u8]>> = LazyLock::new(|| {{"
    )
    .unwrap();
    writeln!(f, "    let mut m = HashMap::new();").unwrap();

    for (name, cubin_path) in &kernels {
        writeln!(
            f,
            "    m.insert(\"{name}\", include_bytes!(\"{}\").as_slice());",
            cubin_path.display()
        )
        .unwrap();
    }

    writeln!(f, "    m").unwrap();
    writeln!(f, "}});").unwrap();
}

fn find_ptxas() -> PathBuf {
    if let Ok(p) = env::var("PTXAS") {
        return PathBuf::from(p);
    }
    if let Ok(cuda) = env::var("CUDA_PATH") {
        let p = PathBuf::from(cuda).join("bin/ptxas");
        if p.exists() {
            return p;
        }
    }
    for path in [
        "/usr/local/cuda/bin/ptxas",
        "/opt/cuda/bin/ptxas",
        "/usr/bin/ptxas",
    ] {
        let p = PathBuf::from(path);
        if p.exists() {
            return p;
        }
    }
    PathBuf::from("ptxas")
}

fn find_cpp() -> PathBuf {
    if let Ok(p) = env::var("CPP") {
        return PathBuf::from(p);
    }
    for path in ["/usr/bin/cpp", "/usr/local/bin/cpp"] {
        let p = PathBuf::from(path);
        if p.exists() {
            return p;
        }
    }
    PathBuf::from("cpp")
}

fn discover_ptx_files(root: &Path) -> Vec<PathBuf> {
    let mut results = Vec::new();
    visit_dir(root, &mut results);
    results.sort();
    results
}

fn visit_dir(dir: &Path, results: &mut Vec<PathBuf>) {
    let Ok(entries) = fs::read_dir(dir) else {
        return;
    };
    for entry in entries.flatten() {
        let path = entry.path();
        if path.is_dir() {
            visit_dir(&path, results);
        } else if path.extension().map_or(false, |e| e == "ptx") {
            // Only include files with .entry directives (actual kernels, not headers)
            if let Ok(content) = fs::read_to_string(&path) {
                if content.contains(".entry") {
                    results.push(path);
                }
            }
        }
    }
}
```

### cudarc Module Loading

The `module.rs` file uses cudarc's `CudaDevice::load_ptx()` method, which internally calls `cuModuleLoadData`. Despite the name, this function accepts both PTX text and cubin ELF binaries -- the CUDA driver distinguishes them by magic number.

```rust
// crates/sm121-kernels/src/module.rs

use std::collections::HashMap;
use std::sync::Arc;
use cudarc::driver::{CudaDevice, CudaFunction};
use cudarc::nvrtc::Ptx;
use crate::error::{Result, SparkError};

include!(concat!(env!("OUT_DIR"), "/embedded_kernels.rs"));

static FUNCTION_CACHE: parking_lot::RwLock<
    Option<HashMap<(usize, String), CudaFunction>>,
> = parking_lot::RwLock::new(None);

fn ensure_cache() {
    let read = FUNCTION_CACHE.read();
    if read.is_some() {
        return;
    }
    drop(read);
    let mut write = FUNCTION_CACHE.write();
    if write.is_none() {
        *write = Some(HashMap::new());
    }
}

pub fn load_kernel(
    dev: &Arc<CudaDevice>,
    kernel_name: &'static str,
    entry_point: &str,
) -> Result<CudaFunction> {
    ensure_cache();

    let cache_key = (dev.ordinal(), format!("{kernel_name}::{entry_point}"));

    // Fast path
    {
        let cache = FUNCTION_CACHE.read();
        if let Some(map) = cache.as_ref() {
            if let Some(func) = map.get(&cache_key) {
                return Ok(func.clone());
            }
        }
    }

    // Slow path: load cubin
    let cubin_bytes = KERNEL_REGISTRY
        .get(kernel_name)
        .ok_or_else(|| SparkError::KernelNotFound(kernel_name.to_string()))?;

    // Ptx::Image accepts raw cubin bytes. cuModuleLoadData dispatches correctly.
    dev.load_ptx(
        Ptx::Image(cubin_bytes.to_vec()),
        kernel_name,
        &[entry_point],
    )?;

    let func = dev
        .get_func(kernel_name, entry_point)
        .ok_or_else(|| SparkError::KernelNotFound(format!("{kernel_name}::{entry_point}")))?;

    // Cache
    {
        let mut cache = FUNCTION_CACHE.write();
        if let Some(map) = cache.as_mut() {
            map.insert(cache_key, func.clone());
        }
    }

    Ok(func)
}
```

// ptx/common/reduction.ptxh
// Warp-level and block-level reduction primitives

#ifndef SPARK_REDUCTION_PTXH
#define SPARK_REDUCTION_PTXH

// Warp-level butterfly reduction for f32 sum.
// Expects %f_val to contain the input value, result left in %f_val.
// Clobbers %f_rtmp.
// Usage: after this macro, lane 0 of each warp has the sum, but
//        all lanes have the result due to butterfly pattern.
//
// WARP_REDUCE_SUM_F32(val_reg, tmp_reg)
#define WARP_REDUCE_SUM_F32(val, tmp) \
    shfl.sync.bfly.b32 tmp, val, 16, 31, FULL_MASK; \
    add.f32 val, val, tmp; \
    shfl.sync.bfly.b32 tmp, val, 8, 31, FULL_MASK; \
    add.f32 val, val, tmp; \
    shfl.sync.bfly.b32 tmp, val, 4, 31, FULL_MASK; \
    add.f32 val, val, tmp; \
    shfl.sync.bfly.b32 tmp, val, 2, 31, FULL_MASK; \
    add.f32 val, val, tmp; \
    shfl.sync.bfly.b32 tmp, val, 1, 31, FULL_MASK; \
    add.f32 val, val, tmp;

// Warp-level butterfly reduction for f32 max.

### smem_swizzle.ptxh

```
// ptx/common/smem_swizzle.ptxh
// XOR-based shared memory swizzling for bank-conflict-free ldmatrix access.
//
// For a row-major matrix in shared memory with row stride S (bytes),
// the swizzled byte offset is:
//   swizzled_addr = base_addr ^ ((row & 7) << 4)
//
// This targets bits [6:4] of the byte address (bank group selection)
// and ensures threads in a warp reading consecutive rows hit different
// bank groups.

#ifndef SPARK_SMEM_SWIZZLE_PTXH
#define SPARK_SMEM_SWIZZLE_PTXH

// SWIZZLE_ADDR(dst_reg, base_reg, row_reg)
// dst_reg (.b32): output swizzled address
// base_reg (.b32): base shared memory byte offset
// row_reg (.b32): row index within the tile
// Clobbers: uses dst_reg as temporary
#define SWIZZLE_ADDR(dst, base, row) \
    and.b32 dst, row, 7; \
    shl.b32 dst, dst, 4; \
    xor.b32 dst, base, dst;

#endif // SPARK_SMEM_SWIZZLE_PTXH
```

### convert.ptxh

```
// ptx/common/convert.ptxh
// Type conversion helpers

#ifndef SPARK_CONVERT_PTXH
#define SPARK_CONVERT_PTXH

// LOG2E constant for exp(x) = 2^(x * log2(e))
#define LOG2E 1.44269504

// Convert exp(x) using base-2 exponential:
//   exp(x) = 2^(x * log2(e))
// EXP_F32(dst, src, tmp)
//   dst (.f32): result
//   src (.f32): input x
//   tmp (.f32): temporary
#define EXP_F32(dst, src, tmp) \
    mul.f32 tmp, src, LOG2E; \
    ex2.approx.f32 dst, tmp;

// Sigmoid: 1 / (1 + exp(-x))
// SIGMOID_F32(dst, src, tmp1, tmp2)
#define SIGMOID_F32(dst, src, tmp1, tmp2) \
    neg.f32 tmp1, src; \
    mul.f32 tmp1, tmp1, LOG2E; \
    ex2.approx.f32 tmp1, tmp1; \
    add.f32 tmp2, tmp1, 1.0; \
    rcp.approx.f32 dst, tmp2;

#endif // SPARK_CONVERT_PTXH
```

#ifndef SPARK_MBARRIER_PTXH

// Initialize an mbarrier for N expected arrivals.

## 3. Hello World Kernel (vector_add)

This kernel validates the full pipeline. Write it first.

```
// ptx/test/vector_add.ptx
//
// Simple vector addition: C[i] = A[i] + B[i]
// Used to validate: PTX authoring -> cpp -> ptxas -> cubin -> embed -> load -> launch

.version 8.5
.target sm_121a
.address_size 64

.visible .entry vector_add(
    .param .u64 param_A,
    .param .u64 param_B,
    .param .u64 param_C,
    .param .u32 param_N
)
{
    .reg .b32 %r<8>;
    .reg .b64 %rd<8>;
    .reg .f32 %f<4>;
    .reg .pred %p<2>;

    // Compute global thread index
    mov.u32         %r0, %ctaid.x;
    mov.u32         %r1, %ntid.x;
    mov.u32         %r2, %tid.x;
    mad.lo.u32      %r3, %r0, %r1, %r2;    // idx = blockIdx.x * blockDim.x + threadIdx.x

    // Bounds check
    ld.param.u32    %r4, [param_N];
    setp.ge.u32     %p0, %r3, %r4;
    @%p0 ret;

    // Load parameters
    ld.param.u64    %rd0, [param_A];
    ld.param.u64    %rd1, [param_B];
    ld.param.u64    %rd2, [param_C];

    // Compute byte offset: idx * 4 (f32)
    mul.wide.u32    %rd3, %r3, 4;

    // Load A[idx] and B[idx]
    add.u64         %rd4, %rd0, %rd3;
    add.u64         %rd5, %rd1, %rd3;
    ld.global.f32   %f0, [%rd4];
    ld.global.f32   %f1, [%rd5];

    // C[idx] = A[idx] + B[idx]
    add.f32         %f2, %f0, %f1;

    // Store result
    add.u64         %rd6, %rd2, %rd3;
    st.global.f32   [%rd6], %f2;

    ret;
}
```

## 4. Flash Attention Kernel Design

### 4.1 Configuration

| Parameter | BF16 d=128 | FP8 d=128 |
|-----------|-----------|-----------|
| Br (Q tile rows) | 128 | 128 |
| Bc (KV tile cols) | 64 | 128 |
| Warps | 8 (256 threads) | 4 (128 threads) |
| WARP_Q | 16 | 32 |
| MMA shape | m16n8k16 | m16n8k32 |
| Q in registers | Yes | Yes |
| K double-buffered SMEM | 2 x 16 KB | 2 x 16 KB |
| V in SMEM | 16 KB | 16 KB |
| Total SMEM | ~48 KB | ~48 KB |
| Regs per thread | ~152 | ~200 |

### 4.2 Register Budget (BF16, 8 warps, WARP_Q=16)

```
Q fragment:      1 row-group * 8 k-chunks * 4 regs = 32 regs
S accumulator:   1 row-group * 8 n-groups * 4 regs = 32 regs
O accumulator:   1 row-group * 16 n-groups * 4 regs = 64 regs
Softmax state:   row_max (2) + row_sum (2)         =  4 regs
K/V operands:    loaded from SMEM per iteration     =  ~4 regs (reused)
Temporaries:     address calculation, predicates     = ~16 regs
                                                     --------
Total:                                               ~152 regs
```

Fits within 256 with headroom for the compiler to spill if needed.

### 4.3 Algorithm (FlashAttention-2 Online Softmax)

```
for each Q-block (outer loop, grid dimension):
    Load Q tile into registers via SMEM + ldmatrix.x4

    Initialize: row_max = -inf, row_sum = 0, O = 0

    for each KV-block (inner loop):
        Load K tile to SMEM (double-buffered)

        // Compute S = Q @ K^T
        for k = 0..d/16:
            for n = 0..Bc/8:
                ldmatrix K fragment from SMEM
                mma.sync.aligned.m16n8k16... S += Q[k] * K[k,n]

        // Apply causal mask (if causal)
        for each S element:
            if kv_pos > q_pos: S[i][j] = -inf

        // Online softmax update
        new_max = max(row_max, rowmax(S))
        correction = exp(old_max - new_max)
        O *= correction
        row_sum *= correction
        P = exp(S - new_max)
        row_sum += rowsum(P)

        // Convert P to bf16 for MMA
        P_bf16 = cvt.rn.bf16.f32(P)

        Load V tile to SMEM

        // Accumulate O += P @ V
        for k = 0..Bc/16:
            for n = 0..d/8:
                ldmatrix V fragment from SMEM
                mma.sync.aligned.m16n8k16... O += P[k] * V[k,n]

    // Finalize
    O /= row_sum   (rcp.approx.f32 + mul)
    Convert O to bf16, store to global memory
    Store LSE = log(row_sum) + row_max
```

### 4.4 Key PTX Sequences

#### Kernel header and thread setup

```ptx
.version 8.5
.target sm_121a
.address_size 64

#include "macros.ptxh"
#include "smem_swizzle.ptxh"
#include "reduction.ptxh"
#include "convert.ptxh"

.shared .align 128 .b8 smem[49152];   // 48 KB

.visible .entry flash_attn_bf16_d128(
    .param .u64 param_Q,
    .param .u64 param_K,
    .param .u64 param_V,
    .param .u64 param_O,
    .param .u64 param_LSE,
    .param .u32 param_seq_len_q,
    .param .u32 param_seq_len_kv,
    .param .u32 param_num_heads,
    .param .u32 param_num_kv_heads,
    .param .f32 param_scale
)
{
    .reg .b32  %r<256>;
    .reg .b64  %rd<64>;
    .reg .f32  %f<128>;
    .reg .b16  %h<16>;
    .reg .pred %p<16>;

    // Thread/warp/block IDs
    mov.u32     %r_tid, %tid.x;              // [0..255]
    mov.u32     %r_q_block, %ctaid.x;        // Q sequence block
    mov.u32     %r_head, %ctaid.y;           // attention head
    mov.u32     %r_batch, %ctaid.z;          // batch

    shr.u32     %r_warp_id, %r_tid, 5;      // warp_id = tid / 32
    and.b32     %r_lane, %r_tid, 31;         // lane_id = tid % 32
    shr.u32     %r_group_id, %r_lane, 2;    // groupID = lane / 4 (MMA row within warp)
    and.b32     %r_tid_in_group, %r_lane, 3; // threadID_in_group = lane % 4

    // ... kernel body follows ...
}
```

#### BF16 MMA instruction

```ptx
    // mma.sync.aligned.m16n8k16.row.col.f32.bf16.bf16.f32
    //
    // A-operand (Q or P): 4 registers (.b32), each holding 2 bf16 values
    //   Thread layout: 8 groups of 4 threads
    //   Group g holds rows [g, g+8] (2 rows)
    //   Thread t in group holds cols [t*4 .. t*4+3] (4 bf16 = 2 regs of bf16x2)
    //   Total: 16 rows x 16 cols (k-dim) -> 4 regs per thread
    //
    // B-operand (K^T or V): 2 registers (.b32)
    //   Thread layout: same 8x4 but transposed
    //   Total: 16 cols (k-dim) x 8 rows (n-dim) -> 2 regs per thread
    //
    // C/D accumulator: 4 registers (.f32)
    //   Same thread mapping as A but for output: 16 rows x 8 cols
    //   Each thread holds 4 f32 values: [row0_col0, row0_col1, row8_col0, row8_col1]

    mma.sync.aligned.m16n8k16.row.col.f32.bf16.bf16.f32
        {%f_d0, %f_d1, %f_d2, %f_d3},
        {%r_a0, %r_a1, %r_a2, %r_a3},
        {%r_b0, %r_b1},
        {%f_c0, %f_c1, %f_c2, %f_c3};
```

#### Loading fragments from SMEM via ldmatrix

```ptx
    // ldmatrix.sync.aligned.m8n8.x4.shared.b16
    // Loads 4 consecutive 8x8 bf16 matrices from shared memory
    // into register fragments suitable for MMA A-operand.
    //
    // Each thread provides a shared memory address.
    // Thread t (lane t) loads from row (t % 8) of the sub-tile.
    // The .x4 variant loads 4 such tiles to fill the m16xk16 operand.

    ldmatrix.sync.aligned.m8n8.x4.shared.b16
        {%r_a0, %r_a1, %r_a2, %r_a3}, [%rd_smem_q_addr];

    // For B-operand (transposed): use .trans variant
    ldmatrix.sync.aligned.m8n8.x2.trans.shared.b16
        {%r_b0, %r_b1}, [%rd_smem_k_addr];
```

#### Online softmax (row-max + rescale)

```ptx
    // Find row max across all S elements for this thread's rows.
    // Each thread owns elements from 2 rows (groupID and groupID+8).
    // First reduce across n-groups within this thread:
    mov.f32     %f_max_r0, 0xff800000;   // -inf
    mov.f32     %f_max_r1, 0xff800000;

    // For each n-group's D-operand, extract row0 and row1 maxes:
    max.f32     %f_max_r0, %f_max_r0, %f_s_n0_r0;  // S[n=0, row=groupID]
    max.f32     %f_max_r1, %f_max_r1, %f_s_n0_r1;  // S[n=0, row=groupID+8]
    // ... repeat for n=1..7 ...

    // Then reduce across the 4 threads sharing each row:
    // Threads with lane%4 == 0,1,2,3 share the same 2 rows.
    shfl.sync.bfly.b32  %f_tmp, %f_max_r0, 1, 0x3, FULL_MASK;
    max.f32     %f_max_r0, %f_max_r0, %f_tmp;
    shfl.sync.bfly.b32  %f_tmp, %f_max_r0, 2, 0x3, FULL_MASK;
    max.f32     %f_max_r0, %f_max_r0, %f_tmp;
    // Same for %f_max_r1

    // Update running max and compute correction
    max.f32     %f_new_max_r0, %f_old_max_r0, %f_max_r0;
    sub.f32     %f_diff_r0, %f_old_max_r0, %f_new_max_r0;
    mul.f32     %f_diff_r0, %f_diff_r0, LOG2E;
    ex2.approx.f32  %f_corr_r0, %f_diff_r0;   // exp(old_max - new_max)

    // Rescale O accumulators
    mul.f32     %f_o_n0_r0, %f_o_n0_r0, %f_corr_r0;
    // ... all O registers for row 0 ...
    mul.f32     %f_rowsum_r0, %f_rowsum_r0, %f_corr_r0;

    // Compute P = exp(S - new_max) and accumulate row_sum
    sub.f32     %f_p, %f_s_elem, %f_new_max_r0;
    mul.f32     %f_p, %f_p, LOG2E;
    ex2.approx.f32  %f_p, %f_p;
    add.f32     %f_rowsum_r0, %f_rowsum_r0, %f_p;

    // Convert P to bf16 for the P@V MMA
    cvt.rn.bf16.f32  %h0, %f_p0;
    cvt.rn.bf16.f32  %h1, %f_p1;
    mov.b32     %r_p_packed, {%h0, %h1};

    // Update stored max
    mov.f32     %f_old_max_r0, %f_new_max_r0;
```

#### Causal masking

```ptx
    // For each S element, check if kv_pos > q_pos
    // q_pos = q_block_start + row_in_tile
    //   row_in_tile = warp_id * 16 + groupID (for row 0)
    //              or warp_id * 16 + groupID + 8 (for row 1)
    // kv_pos = kv_block_start + col_j
    //   col_j = n_group * 8 + tid_in_group * 2 (+ 0 or 1 for the two cols)

    add.u32     %r_qpos_r0, %r_q_block_start, %r_row_r0;
    add.u32     %r_kvpos, %r_kv_block_start, %r_col;

    setp.gt.u32 %p_mask, %r_kvpos, %r_qpos_r0;
    @%p_mask mov.f32 %f_s_elem, 0xff800000;    // -inf
```

#### Epilogue: write O and LSE

```ptx
    // O /= row_sum
    rcp.approx.f32  %f_inv_sum_r0, %f_rowsum_r0;
    mul.f32     %f_o_n0_r0, %f_o_n0_r0, %f_inv_sum_r0;
    // ... all O registers ...

    // Convert O from f32 to bf16
    cvt.rn.bf16.f32  %h_o0, %f_o_n0_r0;
    cvt.rn.bf16.f32  %h_o1, %f_o_n0_r1;

    // Store to global memory (coalesced writes)
    // Thread writes to O[batch, head, q_pos, col]
    // col determined by: n_group * 8 + tid_in_group * 2 + {0,1}
    st.global.v2.b16 [%rd_o_addr], {%h_o0, %h_o1};

    // Write LSE: log(row_sum) + row_max
    // LSE[batch, head, q_pos] = ln(row_sum) + max
    lg2.approx.f32  %f_log_sum, %f_rowsum_r0;
    mul.f32     %f_log_sum, %f_log_sum, 0.693147;  // * ln(2) for natural log
    add.f32     %f_lse, %f_log_sum, %f_old_max_r0;
    st.global.f32 [%rd_lse_addr], %f_lse;
```

### 4.5 SMEM Layout

```
Offset 0:       K_buf[0]   64 rows x 128 cols x 2B = 16,384 bytes
Offset 16384:   K_buf[1]   64 rows x 128 cols x 2B = 16,384 bytes
Offset 32768:   V_buf      64 rows x 128 cols x 2B = 16,384 bytes
                                                      ------
                Total:                                 49,152 bytes (48 KB)
```

K is double-buffered: while computing S with K_buf[i], load next K tile into K_buf[1-i].
V reuses one of the K buffers after K is consumed (V overlaps K_buf[0] after the QK^T computation for that tile is done).

All SMEM accesses use XOR swizzling (see smem_swizzle.ptxh) to avoid bank conflicts during ldmatrix.

## 5. GEMM Kernel Design

### 5.1 FP8 GEMM

| Parameter | Value |
|-----------|-------|
| CTA tile | 128 x 128 |
| K-tile | 64 |
| Warps | 4 (128 threads) |
| Warp tile | 64 x 64 |
| MMA shape | m16n8k32 (FP8) |
| Pipeline stages | 3 |
| SMEM per stage | A: 8KB + B: 8KB = 16KB |
| Total SMEM | 48 KB |
| Regs per thread | ~182 |

#### MMA instruction

```ptx
    mma.sync.aligned.m16n8k32.row.col.f32.e4m3.e4m3.f32
        {%f_d0, %f_d1, %f_d2, %f_d3},
        {%r_a0, %r_a1, %r_a2, %r_a3},
        {%r_b0, %r_b1},
        {%f_c0, %f_c1, %f_c2, %f_c3};
```

#### 3-stage pipeline

```ptx
    // Prologue: prefetch stages 0 and 1
    // [cp.async loads for stage 0]
    cp.async.commit_group;
    // [cp.async loads for stage 1]
    cp.async.commit_group;

GEMM_LOOP:
    // Wait for current compute stage (2 groups outstanding allowed)
    cp.async.wait_group 1;
    bar.sync 0;

    // Issue loads for next stage
    // [cp.async loads for stage (current+2) % 3]
    cp.async.commit_group;

    // Compute: iterate over K-tile, issue MMA instructions
    // For k_group in [0..1]:  (K-tile=64, k_per_mma=32)
    //   ldmatrix A and B fragments from smem[stage_read]
    //   For each (m_group, n_group) in warp tile:
    //     mma.sync.aligned.m16n8k32...

    // Advance stages
    // if more K-tiles remain: goto GEMM_LOOP

    // Epilogue: wait for last stage, final compute, store C
    cp.async.wait_group 0;
    bar.sync 0;
    // [final MMA + store]
```

### 5.2 NVFP4 GEMM

| Parameter | Value |
|-----------|-------|
| CTA tile | 128 x 128 |
| K-tile | 128 (FP4 halves element size) |
| MMA | `kind::mxf8f6f4.block_scale.scale_vec::1X.m16n8k32` on sm_121a |
| SMEM per stage | A: 8KB + scales: 1KB + B: 8KB + scales: 1KB = 18KB |
| Total SMEM (3 stages) | 54 KB |

**Requires**: `sm_121a` target and CUDA 13+ ptxas. The `e2m1` data type represents 4-bit values, and `ue8m0` represents the block scale factors.

```ptx
    // Block-scaled FP4 MMA
    mma.sync.aligned.kind::mxf8f6f4.block_scale.scale_vec::1X.m16n8k32.row.col.f32.e2m1.e2m1.f32.ue8m0
        {%f_d0, %f_d1, %f_d2, %f_d3},
        {%r_a0, %r_a1, %r_a2, %r_a3},
        {%r_b0, %r_b1},
        {%f_c0, %f_c1, %f_c2, %f_c3},
        %r_scaleA, %r_scaleB,
        %r_bidA, %r_tidA,
        %r_bidB, %r_tidB;
```

### 5.3 W4A16 GEMM

Strategy: Load 4-bit packed weights, dequantize to BF16 in SMEM or registers, then use standard BF16 m16n8k16 MMA.

```ptx
    // Unpack 4-bit from 32-bit register (8 values)
    // Extract nibble at position i:
    shr.b32     %r_shifted, %r_packed, %r_shift;   // shift = i * 4
    and.b32     %r_nibble, %r_shifted, 0xF;

    // Convert to bf16: (nibble - zero_point) * scale
    cvt.rn.f32.u32  %f_val, %r_nibble;
    sub.f32     %f_val, %f_val, %f_zero_point;
    mul.f32     %f_val, %f_val, %f_scale;
    cvt.rn.bf16.f32 %h_val, %f_val;
```

| Parameter | Value |
|-----------|-------|
| CTA tile | 128 x 128 |
| K-tile | 64 |
| MMA | m16n8k16 BF16 (after dequant) |
| Pipeline | 2-stage |
| SMEM per stage | Weights 4KB + scales 64B + activations 16KB + dequant buf 16KB = ~36KB |
| Total SMEM | ~72 KB |

## 6. Elementwise Kernel Designs

### 6.1 RMSNorm

One CTA per row. Two-pass: compute sum-of-squares (reduction), then normalize.

```ptx
.visible .entry rmsnorm_bf16(
    .param .u64 param_x,
    .param .u64 param_out,
    .param .u64 param_weight,
    .param .u32 param_hidden_dim,
    .param .f32 param_eps
)
{
    // 256 threads per block, 1 block per row
    // Stage 1: Each thread accumulates partial sum_sq over its elements
    // Stage 2: Warp butterfly reduction (WARP_REDUCE_SUM_F32)
    // Stage 3: Cross-warp reduction via 32-element shared memory array
    // Stage 4: Broadcast rsqrt(mean_sq + eps) to all threads
    // Stage 5: Each thread normalizes and scales its elements

    // Key instructions:
    //   fma.rn.f32 %f_sum, %f_val, %f_val, %f_sum;   (accumulate x^2)
    //   rsqrt.approx.f32 %f_rms_inv, %f_mean_sq;      (inverse RMS)
    //   mul.f32 %f_out, %f_val, %f_rms_inv;
    //   mul.f32 %f_out, %f_out, %f_weight;
}
```

### 6.2 RoPE

### 6.4 GeLU * mul / GeLU-tanh * mul

Same structure as SiLU, different activation:

- **GeLU**: `x * 0.5 * (1 + erf(x * M_SQRT1_2))`
  - PTX: Use polynomial approximation for erf, or `erf` may be available as a PTX intrinsic
- **GeLU-tanh**: `x * 0.5 * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))`
  - PTX: tanh via `exp`: `tanh(z) = 1 - 2/(exp(2z) + 1)`

    pub o_ptr: u64,

pub fn flash_attention(

    params: &FlashAttentionParams,

## 9. Benchmark Harness

```rust
// benches/common.rs

use cudarc::driver::{CudaDevice, CudaStream};
use std::sync::Arc;

pub struct GpuBenchmark {
    pub dev: Arc<CudaDevice>,
}

pub struct BenchResult {
    pub name: String,
    pub median_us: f64,
    pub mean_us: f64,
    pub tflops: f64,
    pub bandwidth_gb_s: f64,
}

impl GpuBenchmark {
    pub fn new() -> Self {
        let dev = CudaDevice::new(0).unwrap();
        Self { dev: Arc::new(dev) }
    }

    pub fn bench<F>(
        &self,
        name: &str,
        warmup: usize,
        iters: usize,
        flops: f64,
        bytes: f64,
        mut f: F,
    ) -> BenchResult
    where
        F: FnMut(&Arc<CudaDevice>),
    {
        // Warmup
        for _ in 0..warmup {
            f(&self.dev);
        }
        self.dev.synchronize().unwrap();

        // Timed runs using CUDA events
        let mut times = Vec::with_capacity(iters);
        for _ in 0..iters {
            let start = self.dev.new_event(Default::default()).unwrap();
            let stop = self.dev.new_event(Default::default()).unwrap();
            start.record(Default::default()).unwrap();
            f(&self.dev);
            stop.record(Default::default()).unwrap();
            stop.synchronize().unwrap();
            let ms = stop.elapsed_ms(&start).unwrap();
            times.push(ms as f64 * 1000.0); // us
        }

        times.sort_by(|a, b| a.partial_cmp(b).unwrap());
        let median = times[iters / 2];
        let mean = times.iter().sum::<f64>() / iters as f64;

        BenchResult {
            name: name.to_string(),
            median_us: median,
            mean_us: mean,
            tflops: if median > 0.0 { flops / (median * 1e-6) / 1e12 } else { 0.0 },
            bandwidth_gb_s: if median > 0.0 { bytes / (median * 1e-6) / 1e9 } else { 0.0 },
        }
    }
}

impl std::fmt::Display for BenchResult {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(
            f,
            "{:<40} median: {:>8.1} us  mean: {:>8.1} us  {:.2} TFLOPS  {:.1} GB/s",
            self.name, self.median_us, self.mean_us, self.tflops, self.bandwidth_gb_s
        )
    }
}
```

#!/usr/bin/env python3
# tests/reference/generate_golden.py

import torch
import numpy as np
from pathlib import Path

OUT = Path(__file__).parent / "data"
OUT.mkdir(exist_ok=True)

def gen_rmsnorm():
    for hidden in [2048, 4096, 8192]:
        x = torch.randn(32, hidden, dtype=torch.bfloat16, device="cuda")
        w = torch.randn(hidden, dtype=torch.bfloat16, device="cuda")
        eps = 1e-5
        rms = torch.sqrt(x.float().pow(2).mean(-1, keepdim=True) + eps)
        out = (x.float() / rms * w.float()).to(torch.bfloat16)
        np.savez(OUT / f"rmsnorm_bf16_h{hidden}.npz",
                 x=x.cpu().numpy(), weight=w.cpu().numpy(),
                 out=out.cpu().numpy(), eps=np.float32(eps))

def gen_silu_mul():
    for d in [2048, 4096]:
        inp = torch.randn(32, 2 * d, dtype=torch.bfloat16, device="cuda")
        gate = inp[..., :d].float()
        up = inp[..., d:].float()
        out = (torch.nn.functional.silu(gate) * up).to(torch.bfloat16)
        np.savez(OUT / f"silu_mul_bf16_d{d}.npz",
                 input=inp.cpu().numpy(), out=out.cpu().numpy())

def gen_rope():
    for d in [64, 128]:
        seq, heads = 512, 32
        x = torch.randn(1, seq, heads, d, dtype=torch.bfloat16, device="cuda")
        # Simple RoPE with base 10000
        pos = torch.arange(seq, device="cuda").float()
        dim_pairs = torch.arange(0, d, 2, device="cuda").float()
        theta = 1.0 / (10000.0 ** (dim_pairs / d))
        angles = pos.unsqueeze(1) * theta.unsqueeze(0)  # [seq, d/2]
        cos_cache = torch.cos(angles)
        sin_cache = torch.sin(angles)

        x_f = x.float()
        x1 = x_f[..., 0::2]
        x2 = x_f[..., 1::2]
        out_f = torch.zeros_like(x_f)
        out_f[..., 0::2] = x1 * cos_cache.unsqueeze(0).unsqueeze(2) - x2 * sin_cache.unsqueeze(0).unsqueeze(2)
        out_f[..., 1::2] = x1 * sin_cache.unsqueeze(0).unsqueeze(2) + x2 * cos_cache.unsqueeze(0).unsqueeze(2)
        out = out_f.to(torch.bfloat16)

        np.savez(OUT / f"rope_bf16_d{d}.npz",
                 x=x.cpu().numpy(), out=out.cpu().numpy(),
                 cos_cache=cos_cache.cpu().numpy(),
                 sin_cache=sin_cache.cpu().numpy())

def gen_flash_attention():
    for dtype_name, dtype in [("bf16", torch.bfloat16)]:
        for causal in [True, False]:
            for seq in [256, 1024]:
                B, H, d = 2, 8, 128
                q = torch.randn(B, H, seq, d, dtype=dtype, device="cuda")
                k = torch.randn(B, H, seq, d, dtype=dtype, device="cuda")
                v = torch.randn(B, H, seq, d, dtype=dtype, device="cuda")
                scale = 1.0 / (d ** 0.5)
                attn = torch.matmul(q.float(), k.float().transpose(-2, -1)) * scale
                if causal:
                    mask = torch.triu(torch.ones(seq, seq, device="cuda"), diagonal=1).bool()
                    attn.masked_fill_(mask, float('-inf'))
                attn = torch.softmax(attn, dim=-1)
                o = torch.matmul(attn, v.float()).to(dtype)
                tag = f"{'causal' if causal else 'noncausal'}_s{seq}"
                np.savez(OUT / f"flash_attn_{dtype_name}_{tag}.npz",
                         q=q.cpu().numpy(), k=k.cpu().numpy(),
                         v=v.cpu().numpy(), o=o.cpu().numpy(),
                         scale=np.float32(scale))

def gen_gemm():
    for m, n, k in [(128, 128, 128), (512, 512, 512), (1024, 4096, 4096)]:
        a = torch.randn(m, k, dtype=torch.bfloat16, device="cuda")
        b = torch.randn(k, n, dtype=torch.bfloat16, device="cuda")
        c = torch.matmul(a.float(), b.float()).to(torch.bfloat16)
        np.savez(OUT / f"gemm_bf16_{m}x{n}x{k}.npz",
                 a=a.cpu().numpy(), b=b.cpu().numpy(), c=c.cpu().numpy())

if __name__ == "__main__":
    print("Generating golden test vectors...")
    gen_rmsnorm()
    print("  rmsnorm done")
    gen_silu_mul()
    print("  silu_mul done")
    gen_rope()
    print("  rope done")
    gen_flash_attention()
    print("  flash_attention done")
    gen_gemm()
    print("  gemm done")
    print("All golden vectors saved to", OUT)

#!/usr/bin/env bash
